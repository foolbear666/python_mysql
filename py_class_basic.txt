# The __init__() Method
# atributes
# method
class people:
  def __init__(self,name,age):
    self.name=name
    self.age=age

  def sayhi(self):
    print("Hi my name is %s, and I'm %s" % (self.name,self.age))

# class instance

someone=people(name='liyang',age=40)

print(someone.sayhi())
print(someone.name)
print(someone.age)

someone.age=20
print(someone.age)

someone.name='sam'
print(someone.name)

print type(someone)
print type(people)

# private attributes
class people:
  def __init__(self,name,age):
    self.__name=name
    self.__age=age

  def sayhi(self):
    print("Hi my name is %s, and I'm %s" % (self.__name,self.__age))

# class instance

someone=people(name='sam',age=20)
print(someone.name)
print(someone.age)
